⚙️ script.js

Interatividade e lógica da aplicação

// Função para calcular economia de água
function calcularEconomia() {
    let area = document.getElementById('area').value;
    let resultado = document.getElementById('resultado');
    
    if(area && area > 0) {
        let economia = area * 1.5; // litros por hectare
        resultado.innerText = `Com práticas sustentáveis, 
        você economiza ${economia} litros de água por dia.`;
    } else {
        resultado.innerText = 'Por favor, insira uma área válida (hectares).';
    }
}

// Evento de clique
document.getElementById('calcularBtn')?.addEventListener('click', calcularEconomia);

Publicando no GitHub e Configurando GitHub Pages

1

Criando o Repositório

  1. Faça login no GitHub
  2. Clique em "New repository"
  3. Nome: Agrinho
  4. Escolha Público
  5. Clique em "Create repository"
2

Enviando o Projeto Local

git init
git add .
git commit -m "Primeira versão do projeto Agrinho"
git remote add origin https://github.com/SEU_USUARIO/Agrinho.git
git branch -M main
git push -u origin main
3

Ativando GitHub Pages

  1. Vá em Settings → Pages
  2. Selecione main branch
  3. Pasta: /(root)
  4. Clique em Save
  5. Seu site estará em: https://SEU_USUARIO.github.io/Agrinho/
4

Configurando o Link no "About"

  1. Na página do repositório, clique em About (engrenagem)
  2. Marque "Use your GitHub Pages website"
  3. Cole o link do GitHub Pages
  4. Adicione a tag agrinho nos Topics

Regras Importantes da Subcategoria 3 (Front-End)

✅ O que é Obrigatório

  • Arquivos index.html, style.css, script.js separados
  • Repositório público com tag agrinho
  • README.md com instruções
  • Link do GitHub Pages funcional
  • Sem frameworks (HTML/CSS/JS puro)
  • Código indentado e comentado
  • Imagens próprias ou com créditos

❌ O que Causa Desclassificação

  • Projetos idênticos entre estudantes
  • Merge de repositórios ou código copiado
  • CSS/JS inline ou interno
  • Ausência da tag agrinho
  • Link do GitHub Pages não funcionante
  • Conteúdo não alinhado ao tema

// Função para calcular economia de água
function calcularEconomia() {
    // Obtém o valor do campo de entrada
    let areaInput = document.getElementById('area');
    let resultadoParagrafo = document.getElementById('resultado');
    let area = parseFloat(areaInput.value);
    
    // Validação
    if (isNaN(area) || area <= 0) {
        resultadoParagrafo.innerText = '❌ Por favor, insira uma área válida.';
        return;
    }
    
    // Cálculo: 1.500 litros por hectare por dia
    let economiaLitros = area * 1500;
    resultadoParagrafo.innerHTML = 
        `💧 Você economiza ${economiaLitros.toLocaleString()} litros por dia!`;
}

// Adiciona evento ao botão
const botao = document.getElementById('calcularBtn');
if (botao) {
    botao.addEventListener('click', calcularEconomia);
}
